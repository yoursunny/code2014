一 Verilog设计文件

波形发生模块 wavegen.v

module wavegen(output reg d);
    initial
    begin
       d <= #0  1'b0;
       d <= #2  1'b1;
       d <= #3  1'b0;
       d <= #12 1'b1;
       d <= #22 1'b0;
       d <= #24 1'b1;
       d <= #27 1'b0;
       d <= #32 1'b1;
    end
endmodule

测试平台

`timescale 10 ns / 1 ns
`include "wavegen.v"

module wavegen_testbench;
    wire d;
    wavegen w(d);
endmodule
