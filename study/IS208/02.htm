8:3编码器 Encoder8x3.v

module Encoder8x3(output reg [2:0] code, input [7:0] data);
    always @(data)
    case (data)
        8'b00000001: code=3'o0;
        8'b00000010: code=3'o1;
        8'b00000100: code=3'o2;
        8'b00001000: code=3'o3;
        8'b00010000: code=3'o4;
        8'b00100000: code=3'o5;
        8'b01000000: code=3'o6;
        8'b10000000: code=3'o7;
        default:     code=3'ox;
    endcase
endmodule
