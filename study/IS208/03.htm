4位超前进位全加器 fulladder4b.v

module fulladder4b(output [3:0] sum, output c_out, input [3:0] a, input [3:0] b, input cin);
    wire [3:0] g;
    wire [3:0] p;
    wire [3:-1] c;
    wire [3:0] s;
    assign g=a&b;
    assign p=a^b;
    assign c[-1]=cin;
    assign s[3:0]=p[3:0]^c[2:-1];
    assign c[0]=g[0]|(p[0]&c[-1]);
    assign c[1]=g[1]|(p[1]&g[0])|(p[1]&p[0]&c[-1]);
    assign c[2]=g[2]|(p[2]&g[1])|(p[2]&p[1]&g[0])|(p[2]&p[1]&p[0]&c[-1]);
    assign c[3]=g[3]|(p[3]&g[2])|(p[3]&p[2]&g[1])|(p[3]&p[2]&p[1]&g[0])|(p[3]&p[2]&p[1]&p[0]&c[-1]);
    assign sum=s;
    assign c_out=c[3];
endmodule
