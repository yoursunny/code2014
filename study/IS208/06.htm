一 Verilog设计文件

BCD-余3转码 BCD2Excess3.v

module BCD2Excess3(output reg [3:0] dat_out, input [3:0] dat_in);
    always @*
    if ((dat_in>=4'b0000) & (dat_in<=4'b1001))
        dat_out = dat_in + 4'b0011;
    else
        dat_out = 4'bxxxx;
endmodule

测试平台

`include "BCD2Excess3.v"
module tb_BCD2Excess3;
    reg [3:0] dat_in;
    wire [3:0] dat_out;
    BCD2Excess3 be(dat_out,dat_in);
    initial
    begin
        dat_in=4'b0000;
        $monitor("decimal=","%d",dat_in," BCD=","%b",dat_in," Excess3=","%b",dat_out);
    end
    always
        #1 dat_in=dat_in+4'b0001;
endmodule
