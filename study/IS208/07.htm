一 Verilog设计文件

comb_Y1.v

module comb_Y1(output Y,input A,input B,input C);
    wire [2:0] I;
    assign I={A,B,C};
    assign Y=|{I==1,I==2,I==4};
endmodule

测试平台

`include "comb_Y1.v"
module tb_comb_Y1;
    reg A;
    reg B;
    reg C;
    wire Y;
    comb_Y1 c(Y,A,B,C);
    initial
    begin
        {A,B,C}=3'b000;
        $monitor("A=",A," B=",B," C=",C," Y=",Y);
    end
    always
        #1 {A,B,C}={A,B,C}+3'b001;
endmodule

comb_Y2.v

module comb_Y2(output Y,input A,input B,input C,input D);
    wire [3:0] I;
    assign I={A,B,C,D};
    assign Y=|{I==4,I==5,I==6,I==7,I==11,I==12,I==13};
endmodule

测试平台

`include "comb_Y2.v"
module tb_comb_Y2;
    reg A;
    reg B;
    reg C;
    reg D;
    wire Y;
    comb_Y2 c(Y,A,B,C,D);
    initial
    begin
        {A,B,C,D}=4'b0000;
        $monitor("A=",A," B=",B," C=",C," D=",D," Y=",Y);
    end
    always
        #1 {A,B,C,D}={A,B,C,D}+4'b0001;
endmodule
