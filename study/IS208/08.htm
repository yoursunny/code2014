一 Verilog设计文件

ones_count.v

module ones_count(output [3:0] count,input [7:0] dat_in);
    assign count=dat_in[7]+dat_in[6]+dat_in[5]+dat_in[4]+dat_in[3]+dat_in[2]+dat_in[1]+dat_in[0];
endmodule

测试平台

`include "ones_count.v"
module tb_ones_count;
    reg [7:0] dat_in;
    wire [3:0] count;
    ones_count o(count,dat_in);
    initial
    begin
        dat_in=8'b00000000;
        $monitor("dat_in=%b",dat_in," count=%d",count);
    end
    always
        #1 dat_in=dat_in+1;
endmodule
