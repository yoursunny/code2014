一 Verilog设计文件

comb_str.v

`include "mux2x1.v"
module comb_str(output y,input sel,input A,input B,input C,input D);
    wire in0;
    wire in1;
    nand U1(in0,A,B);
    nand U2(in1,C,D);
    mux2x1 mux(y,sel,{in1,in0});
endmodule

测试平台

`include "comb_str.v"
module tb_comb_str;
    reg A;
    reg B;
    reg C;
    reg D;
    reg sel;
    wire y;
    comb_str c(y,sel,A,B,C,D);
    initial
    begin
        {A,B,C,D,sel}=5'b00000;
        $monitor("A=",A," B=",B," C=",C," D=",D," sel=",sel," y=",y);
    end
    always
        #1 {A,B,C,D,sel}={A,B,C,D,sel}+1;
endmodule
