一 Verilog设计文件

计数器 counter8b_updown.v

module counter8b_updown(output reg [7:0] count,input clk,input reset,input dir);
    always @(posedge clk,reset)
    begin
        if (reset)
            count<=8'b00000000;
        else
            if (dir)
                count<=count+8'b00000001;
            else
                count<=count-8'b00000001;
    end
endmodule

测试平台

`include "counter8b_updown.v"
module tb_counter8b_updown;
    wire [7:0] count;
    reg clk;
    reg reset;
    reg dir;
    counter8b_updown n(count,clk,reset,dir);
    initial
    begin
        clk=0'b0;
        reset=0'b1;
        dir=0'b1;
    end
    always
        #1 clk=~clk;
    initial
    begin
        #10  reset=0'b0;
        #591 reset=0'b1;
        #2   reset=0'b0;
        #7   dir=0'b0;
    end
endmodule
