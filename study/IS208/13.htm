计数器 dec_counter.v

module dec_counter(output reg [3:0] count,input clk,input reset);
    always @(posedge clk)
    begin
        if (!reset)
            count<=4'b0000;
        else
            if (count==4'b1001)
                count<=4'b0000;
            else
                count<=count+4'b0001;
    end
endmodule

`include "dec_counter.v"
module tb_dec_counter;
    wire [3:0] count;
    reg clk;
    reg reset;
    dec_counter n(count,clk,reset);
    initial
    begin
        clk=1'b0;
        reset=1'b0;
    end
    always
        #2 clk=~clk;
    initial
    begin
        #5   reset=1'b1;
        #95  reset=1'b0;
        #10  reset=1'b1;
    end
endmodule
