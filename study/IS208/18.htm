一 Verilog设计文件

地址译码 decoder16b.v

module decoder16b(output [7:0] segment,input [15:0] din);
    genvar j;
    generate for (j=0;j<8;j=j+1)
    begin
        assign segment[j]=(din[15-:3]==j);
    end
    endgenerate
endmodule
